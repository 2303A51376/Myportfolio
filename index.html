import copy class WebPageTemplate: def __init__(self, title, header, footer, content_blocks): self.title = title self.header = header self.footer = footer self.content_blocks = content_blocks # list of content sections def clone(self): return copy.deepcopy(self) def update_content(self, new_blocks): self.content_blocks = new_blocks def display(self): print(f"📄 Page Title: {self.title}") print(f"🔝 Header: {self.header}") print("📦 Content Blocks:") for block in self.content_blocks: print(f" - {block}") print(f"🔚 Footer: {self.footer}") print("-" * 50) # 🔹 Base Template base_template = WebPageTemplate( title="Base Page", header="Welcome to Our Website", footer="© 2025 WebLaunch Inc.", content_blocks=["Intro Section", "Feature Highlights", "Call to Action"] ) # 🔹 Clone for Home Page home_page = base_template.clone() home_page.title = "Home" home_page.update_content(["Hero Banner", "Latest News", "Newsletter Signup"]) # 🔹 Clone for About Page about_page = base_template.clone() about_page.title = "About Us" about_page.update_content(["Company History", "Team Profiles", "Mission Statement"]) # 🔹 Display Pages print("✅ Home Page:") home_page.display() print("✅ About Page:") about_page.display() print("✅ Base Template (Unchanged):") base_template.display()
